# Chapter 1, exercise 11
# The two methods perform the same operation with the same ending result.
__________________________________________________________________________

>>> phrase1 = (['What', 'a', 'great', 'morning', '!'])

>>> phrase2 = (['Ending', 'the', 'day', 'with', 'a', 'party', '.'])

>>> len(phrase1 + phrase2)
12

>>> len(phrase1) + len(phrase2)
12
